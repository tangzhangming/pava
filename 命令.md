# 初始化项目
pava pkg init com.my.coolapp
# 编译
pava compile src/Main.pava src/Greeter.pava
# 运行
java -cp target/classes com.my.coolapp.Main
# 输出: Hello from Greeter
#       Hello from com.my.coolapp
# 打包（瘦包）
pava build
# 打包（胖包，包含 vendor）
pava build --fat
# 运行 JAR
java -jar target/com.my.coolapp-1.0.0.jar
